175 
176  if(compteRebours == nullptr)

179  compteRebours->setInterval(TOUTES_LES_SECONDES);

183  }
184 
185  mettreAJourAffichageDuree();
186  compteRebours->start();
187  pomodoro = true;
188  ui->boutonModeCompteARebours->setEnabled(false);

193 void Pomodoro::mettreAJourAffichageDuree()
194 {
195  if(pause)
196  return;
197 
198  QString temps;
199  if(ui->boutonModeCompteARebours->isChecked())
200  {
201  // Vérifier si le temps restant est écoulé
202  if(tempsRestant <= 0)
203  {
204  arreterPomodoro();
205  }
206 
207  // Calculer les minutes et les secondes
208  int minutes = tempsRestant / 60;
209  int secondes = tempsRestant % 60;
210 
211  // Convertir les minutes et les secondes en chaînes de caractères avec le format "mm:ss"
212  temps = QString("%1:%2").arg(QString::number(minutes).rightJustified(2, '0'),
213  QString::number(secondes).rightJustified(2, '0'));
214 
215  ui->affichageDuree->display(temps);
216  qDebug() << Q_FUNC_INFO << "CompteARebours" << temps;
217 
218  // Réduire le temps restant d'une seconde
219  tempsRestant -= 1;
220  }
221  else if(ui->boutonModeChronometre->isChecked())
222  {
223  // Vérifier si le temps a atteint la durée
224 #ifdef DEMO
225  if(tempsRestant >= DUREE_POMODORO)
226 #else
227  if(tempsRestant >= DUREE_POMODORO * 60)
228 #endif
229  {
230  arreterPomodoro();
231  }
232 
233  // Mettre à jour l'affichage du chronomètre
234  int minutes = tempsRestant / 60;
235  int secondes = tempsRestant % 60;
236  temps = QString("%1:%2").arg(QString::number(minutes).rightJustified(2, '0'),
237  QString::number(secondes).rightJustified(2, '0'));
238 
239  ui->affichageDuree->display(temps);
240  qDebug() << Q_FUNC_INFO << "Chronometre" << temps;
241 
242  // Augmenter le temps restant de 1 seconde
243  tempsRestant += 1;
244  }
245 }
